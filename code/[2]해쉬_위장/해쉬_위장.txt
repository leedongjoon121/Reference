import java.util.*;
class Solution {
    public int solution(String[][] clothes) {
        int answer = 1;
        int row = clothes.length;
        HashMap<String,Integer> map = new HashMap<String,Integer>();
        
        for(int i = 0; i< row; i++){
            String str = clothes[i][1];
            map.put(str,map.getOrDefault(str,1)+1); // 안입는 경우도 있기 때문에 기본이 2 부터 시작이다.
        }        
        
       for( String key : map.keySet()){
            int num = map.get(key);
            answer *= num;
        }
         
        return answer-1;
    }
}
