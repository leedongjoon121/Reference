```
알파벳 소문자로 이루어진 N개의 단어가 들어오면 아래와 같은 조건에 따라 정렬하는 프로그램을 작성하시오.

길이가 짧은 것부터
길이가 같으면 사전 순으로
```

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.*;



public class Main  {
	
	public static void main(String[] args) throws IOException  {
    	
	 BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
	 StringBuilder sb = new StringBuilder(); // 출력시 스트링 빌더를 사용한다.
	 int N = Integer.parseInt(br.readLine());
	 ArrayList<String> arrList = new ArrayList<>();
	 HashSet<String> hs = new HashSet<>();
	 
	
	 
	 for(int i = 0 ; i< N; i++) {
		String input = br.readLine();
		if(!hs.contains(input)) { // hashSet이 중복 제거 체
			 arrList.add(input);	 // 없는 얘만 넣음 
		}
		hs.add(input);
		 
	 }
	 
	 Collections.sort(arrList, new Comparator<String>(){

		@Override
		public int compare(String o1, String o2) {
			
			if(o1.length()==o2.length()) {
				return o1.compareTo(o2);
			}
			return o1.length()-o2.length(); // o1이 더 크다면 자동으로 양수가 나오면서 오름차순이 되고  
			                               // o1이 더 작다면 자동으로 음수가 나와서 내림차순이 된다. 
			
		
		}
	 });
	 
	   for(int i = 0; i<arrList.size(); i++) {
		   sb.append(arrList.get(i)).append('\n');
	   }
	   System.out.println(sb);
	  
	}	
	
}
```
